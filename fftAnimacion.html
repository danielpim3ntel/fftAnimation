Transformada rapida de Fourier

def FFT(P):

# P - [p0, p1, ... , pn-1] representacion de coeficientes

n = len(P)

if (n == 1)
 return(P)

w = e^(2pi*i/2)

Pe = [p0, p2, ... , pn-2]

Po = [p1, p3, ... , pn-1]

ye = FFT(Pe)

yo = FFT(Po)

y = [0] * n

for j in range (n/2):
  y[j] = ye[j] + wj * yo[j]
  y[j + n/2] = ye[j] - wj * yo[j]

return y

Multiplicacion de valores

def multiplicacion(Pe, Po):

# res - [res(0), res(11), ... , res(n-1)] representacion de valores (resultado de multiplicacion)

res = [0] * n

for j in range (n/2):
  res[i] = multiplicar(Pe[i], Po[i])

return res

Transformada inversa rapida de Fourier

def IFFT(P):

# P - [p(w0), p(w1), ... , p(wn-1)] representacion de valores

n = len(P)

if (n == 1)
 return(P)

w = (1/n) * e^(2pi*i/2)

Pe = [p0, p2, ... , pn-2]

Po = [p1, p3, ... , pn-1]

ye = IFFT(Pe)

yo = IFFT(Po)

y = [0] * n

for j in range (n/2):
  y[j] = ye[j] + wj * yo[j]
  y[j + n/2] = ye[j] - wj * yo[j]

return y
