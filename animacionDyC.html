Transformada rápida de Fourier

def FFT(P):

 n = len(P)

 if (n == 1):
  return(P)

 wi = e^(2pi*i/2)

 Pe = [p0, p2, ... , pn-2]

 Po = [p1, p3, ... , pn-1]

 ye = FFT(Pe)

 yo = FFT(Po)

 y = [0] * n

 for j in range (n/2):
   y[j] = ye[j] + wj * yo[j]
   y[j + n/2] = ye[j] - wj * yo[j]

 return y

def IFFT(P):

 n = len(P)

 if (n == 1):
  return(P)

 wi = e^(-2pi*i/2)

 Pe = [p0, p2, ... , pn-2]

 Po = [p1, p3, ... , pn-1]

 ye = IFFT(Pe)

 yo = IFFT(Po)

 y = [0] * n

 for j in range (n/2):
   y[j] = ye[j] + wj * yo[j]
   y[j + n/2] = ye[j] - wj * yo[j]

 return y
